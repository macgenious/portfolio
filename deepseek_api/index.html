Sample Code


# AI Chatbot implementation with context management
from fastapi import FastAPI, WebSocket, WebSocketDisconnect
from pydantic import BaseModel
import google.generativeai as genai
import os
from typing import List, Dict

# Initialize FastAPI app
app = FastAPI()

# Configure Gemini API
genai.configure(api_key=os.environ["GEMINI_API_KEY"])
model = genai.GenerativeModel('gemini-pro')

# Chat session management
class ChatSession:
    def __init__(self, session_id: str):
        self.session_id = session_id
        self.chat = model.start_chat(history=[])
        self.context = {}
    
    async def send_message(self, message: str) -> str:
        # Enhance message with context if available
        enhanced_message = self._add_context_to_message(message)
        
        # Get response from model
        response = await self.chat.send_message_async(enhanced_message)
        
        # Extract and store any new context
        self._update_context(message, response.text)
        
        return response.text
    
    def _add_context_to_message(self, message: str) -> str:
        if not self.context:
            return message
            
        context_str = "\n\nContext:\n" + "\n".join(
            [f"{k}: {v}" for k, v in self.context.items()]
        )
        return message + context_str
    
    def _update_context(self, message: str, response: str):
        # Simple context extraction logic
        # In a real app, this would be more sophisticated
        if "my name is" in message.lower():
            name = message.lower().split("my name is")[1].strip()
            self.context["user_name"] = name

# Active chat sessions
active_sessions: Dict[str, ChatSession] = {}

# WebSocket endpoint for real-time chat
@app.websocket("/ws/{session_id}")
async def websocket_endpoint(websocket: WebSocket, session_id: str):
    await websocket.accept()
    
    # Create or retrieve session
    if session_id not in active_sessions:
        active_sessions[session_id] = ChatSession(session_id)
    
    session = active_sessions[session_id]
    
    try:
        while True:
            # Receive message from client
            message = await websocket.receive_text()
            
            # Process message and get response
            response = await session.send_message(message)
            
            # Send response back to client
            await websocket.send_text(response)
    except WebSocketDisconnect:
        # Clean up on disconnect
        if session_id in active_sessions:
            del active_sessions[session_id]
