Sample Code


import javax.swing.*;
import java.awt.*;
import java.awt.event.*;
import java.util.*;

public class AvoidTheDots extends JFrame {
    public static void main(String[] args) {
        new AvoidTheDots();
    }

    public AvoidTheDots() {
        setTitle("Avoid the Dots");
        setDefaultCloseOperation(EXIT_ON_CLOSE);
        add(new GamePanel());
        pack();
        setVisible(true);
    }
}

class GamePanel extends JPanel implements ActionListener {
    private int playerX = 200, playerY = 300, score = 0;
    private ArrayList dots = new ArrayList<>();
    private Random rand = new Random();
    private Timer timer = new Timer(30, this);

    public GamePanel() {
        setPreferredSize(new Dimension(400, 400));
        addKeyListener(new KeyAdapter() {
            public void keyPressed(KeyEvent e) {
                int key = e.getKeyCode();
                if (key == KeyEvent.VK_LEFT) playerX -= 10;
                if (key == KeyEvent.VK_RIGHT) playerX += 10;
                if (key == KeyEvent.VK_UP) playerY -= 10;
                if (key == KeyEvent.VK_DOWN) playerY += 10;
                repaint();
            }
        });
        setFocusable(true);
        timer.start();
    }

    public void actionPerformed(ActionEvent e) {
        if (rand.nextInt(10) < 3) dots.add(new Point(rand.nextInt(400), 0));
        for (int i = dots.size()-1; i >= 0; i--) {
            Point dot = dots.get(i);
            dot.y += 5;
            if (dot.y > 400) dots.remove(i);
            else if (new Rectangle(playerX, playerY, 20, 20).contains(dot)) {
                timer.stop();
                JOptionPane.showMessageDialog(this, "Game Over! Score: " + score);
                System.exit(0);
            }
        }
        score++;
        repaint();
    }

    protected void paintComponent(Graphics g) {
        super.paintComponent(g);
        g.setColor(Color.BLUE);
        g.fillRect(playerX, playerY, 20, 20);
        g.setColor(Color.RED);
        for (Point dot : dots) g.fillOval(dot.x, dot.y, 10, 10);
        g.setColor(Color.BLACK);
        g.drawString("Score: " + score, 10, 20);
    }
}
