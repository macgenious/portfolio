Python Pokedex API

Project Overview

The Python Pokedex API is a comprehensive RESTful API built with Python and Flask that provides detailed information about Pokémon. This project demonstrates API development, database integration, and efficient data retrieval techniques while showcasing clean code architecture and documentation practices.

Key Features

  • Complete Pokémon database with detailed information on all generations
  • RESTful API endpoints with comprehensive documentation
  • Advanced filtering and search capabilities
  • Authentication system with rate limiting
  • Caching mechanisms for improved performance
  • Interactive API testing interface

Technologies Used

Python Flask SQLAlchemy RESTful API JWT Swagger Redis

Sample Code


# Define Pokemon model using SQLAlchemy ORM
class Pokemon(db.Model):
    __tablename__ = 'pokemon'
    
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(100), nullable=False)
    type_primary = db.Column(db.String(50), nullable=False)
    type_secondary = db.Column(db.String(50))
    hp = db.Column(db.Integer, nullable=False)
    attack = db.Column(db.Integer, nullable=False)
    defense = db.Column(db.Integer, nullable=False)
    speed = db.Column(db.Integer, nullable=False)
    generation = db.Column(db.Integer, nullable=False)
    legendary = db.Column(db.Boolean, default=False)
    
    def __repr__(self):
        return f""
    
    def to_dict(self):
        return {
            'id': self.id,
            'name': self.name,
            'types': [
                self.type_primary,
                self.type_secondary
            ] if self.type_secondary else [self.type_primary],
            'stats': {
                'hp': self.hp,
                'attack': self.attack,
                'defense': self.defense,
                'speed': self.speed
            },
            'generation': self.generation,
            'legendary': self.legendary
        }

# API endpoint to get Pokemon by ID
@app.route('/api/pokemon/', methods=['GET'])
@jwt_required
@cache.cached(timeout=300)
def get_pokemon(pokemon_id):
    pokemon = Pokemon.query.get_or_404(pokemon_id)
    return jsonify(pokemon.to_dict())
